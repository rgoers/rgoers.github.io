185     @Value("${gitUserName")
186     private String gitUserName;
187 
188     @Value("${gitPassword:#{null}}")
189     private String gitPassword;
190 
191     @Value("${gitPassPhrase:#{null}}")
192     private String gitPassPhrase;
193 
194     @Value("${localRepoUrl:#{null}}")
195     private String localRepoUrl;
196 
197     @Value("${privateKeyPath:#{null}}")
198     private String privateKeyPath;
199 
200     @Value("${remoteRepoUrl}")
201     private String remoteRepoUrl;
202 
203     @Value("${remoteRepoCatalogPath:#{null}}")
204     private String remoteRepoCatalogPath;

206     @Bean
207     public CatalogDao catalogDao() {
208         GitCatalogDao dataSource = new GitCatalogDao();
209         if (isNotBlank(gitUserName) && isNotBlank(gitPassword)) {
210             dataSource.setCredentialsProvider(new UsernamePasswordCredentialsProvider(gitUserName, gitPassword));
211         }
212         if (isNotBlank(remoteRepoUrl)) {
213             try {
214                 URI uri = new URI(remoteRepoUrl);
215                 if (uri.getScheme().equalsIgnoreCase("SSH")) {
216                     TransportConfigCallback transportConfigCallback = new TransportConfigCallback() {
217                         final SshSessionFactory sshSessionFactory = new JschConfigSessionFactory() {
218                             @Override
219                             protected JSch createDefaultJSch( FS fs ) throws JSchException {
220                                 JSch defaultJSch = super.createDefaultJSch( fs );
221                                 if (isNotBlank(privateKeyPath)) {
222                                     defaultJSch.addIdentity(privateKeyPath);
223                                 }
224                                 return defaultJSch;
225                             }
226 
227                             @Override
228                             protected void configure(OpenSshConfig.Host host, Session session) {
229                                 session.setConfig("StrictHostKeyChecking", "no");
230                                 if (isNotBlank(gitPassPhrase)) {
231                                     session.setUserInfo(new UserInfo() {
232                                         @Override
233                                         public String getPassphrase() {
234                                             return gitPassPhrase;
235                                         }
236 
237                                         @Override
238                                         public String getPassword() {return null;}
239 
240                                         @Override
241                                         public boolean promptPassword(String message) {return false;}
242 
243                                         @Override
244                                         public boolean promptPassphrase(String message) {return true;}
245 
246                                         @Override
247                                         public boolean promptYesNo(String message) {return false;}
248 
249                                         @Override
250                                         public void showMessage(String message) {}
251                                     });
252 
253                                 }
254                             }
255                         };
256                         @Override
257                         public void configure(Transport transport) {
258                             SshTransport sshTransport = ( SshTransport )transport;
259                             sshTransport.setSshSessionFactory( sshSessionFactory );
260 
261                         }
262                     };
263                     dataSource.setTransportConfigCallback(transportConfigCallback);
264                 }
265             } catch (URISyntaxException ex) {
266                 LOGGER.error("Invalid URI {}:", remoteRepoUrl, ex);
267             }
268         } else {
269             LOGGER.error("No remote repo URL provided.");
270         }
271 
272         if (isNotBlank(localRepoUrl)) {
273             dataSource.setLocalRepoPath(localRepoUrl);
274         } else {
275             String localRepoPath = System.getProperty("java.io.tmpdir") + "/audit/catalog";
276             File file = new File(localRepoPath);
277             File parent = file.getParentFile();
278             parent.mkdirs();
279             dataSource.setLocalRepoPath(localRepoPath);
280         }
281 
282         dataSource.setRemoteRepoUri(remoteRepoUrl);
283         if (isNotBlank(remoteRepoCatalogPath)) {
284             dataSource.setCatalogPath(remoteRepoCatalogPath);
285         }
286         return dataSource;
287     }

289     public SpringResourceTemplateResolver templateResolver(){
290         // SpringResourceTemplateResolver automatically integrates with Spring's own
291         // resource resolution infrastructure, which is highly recommended.
292         SpringResourceTemplateResolver templateResolver = new SpringResourceTemplateResolver();
293         templateResolver.setApplicationContext(this.applicationContext);
294         templateResolver.setPrefix("/WEB-INF/templates/");
295         templateResolver.setSuffix(".html");
296         // HTML is the default value, added here for the sake of clarity.
297         templateResolver.setTemplateMode(TemplateMode.HTML);
298         // Template cache is true by default. Set to false if you want
299         // templates to be automatically updated when modified.
300         templateResolver.setCacheable(true);
301         return templateResolver;
302     }
303 
304     public SpringTemplateEngine templateEngine(){
305         // SpringTemplateEngine automatically applies SpringStandardDialect and
306         // enables Spring's own MessageSource message resolution mechanisms.
307         SpringTemplateEngine templateEngine = new SpringTemplateEngine();
308         templateEngine.setTemplateResolver(templateResolver());
309         // Enabling the SpringEL compiler with Spring 4.2.4 or newer can
310         // speed up execution in most scenarios, but might be incompatible
311         // with specific cases when expressions in one template are reused
312         // across different data types, so this flag is "false" by default
313         // for safer backwards compatibility.
314         templateEngine.setEnableSpringELCompiler(true);
315         return templateEngine;
316     }

318     @Bean
319     public ViewResolver thymeleafViewResolver(){
320         ThymeleafViewResolver viewResolver = new ThymeleafViewResolver();
321         viewResolver.setTemplateEngine(templateEngine());
322         // NOTE 'order' and 'viewNames' are optional
323         viewResolver.setOrder(1);
324         viewResolver.setViewNames(new String[] {"products", "categories", "events", "attributes"});
325         return viewResolver;
326     }
