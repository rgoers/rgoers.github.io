1   package org.apache.logging.log4j.core.lookup;
2   
3   import java.util.Map;
4   import java.util.Properties;
5   
6   /**
7    * {@link RuntimeStrSubstitutor} is a {@link StrSubstitutor} which only supports evaluation of top-level lookups.
8    */
9   public final class RuntimeStrSubstitutor extends StrSubstitutor {
10  
11      public RuntimeStrSubstitutor() {
12      }
13  
14      public RuntimeStrSubstitutor(final Map<String, String> valueMap) {
15          super(valueMap);
16      }
17  
18      public RuntimeStrSubstitutor(final Properties properties) {
19          super(properties);
20      }
21  
22      public RuntimeStrSubstitutor(final StrLookup lookup) {
23          super(lookup);
24      }
25  
26      public RuntimeStrSubstitutor(final StrSubstitutor other) {
27          super(other);
28      }
29  
30      @Override
31      boolean isRecursiveEvaluationAllowed() {
32          return false;
33      }
34  
35      @Override
36      void setRecursiveEvaluationAllowed(final boolean recursiveEvaluationAllowed) {
37          throw new UnsupportedOperationException(
38                  "recursiveEvaluationAllowed cannot be modified within RuntimeStrSubstitutor");
39      }
40  
41      @Override
42      public String toString() {
43          return "RuntimeStrSubstitutor{" + super.toString() + "}";
44      }
45  }
